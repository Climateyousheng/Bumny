#!/bin/sh
#
# This script runs the User
# Interface entry system client

# Check for -display option
if test "$1" = "-display"
then
export DISPLAY=$2
fi

# The location of the software
export UI_BASE=${UI_BASE:-@ghui_dir@}

# The initial letter for new experiments
UI_INITIAL=@initial_letter@

# The remote shell command
export REMOTE_SHELL_COMMAND=@remote_shell_command@

# Name of wish executable
export WISH_EXE=${WISH_EXE:-@wish_exe@}

# Start the application
$UI_BASE/bin/ghui @app_name@ $UI_INITIAL
